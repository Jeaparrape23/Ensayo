git clone
git log
git log --oneline


git restore
	para volver y demas
git pull y push
	para descargar y subir cambios

git status
    revisa que cosas no hemos cambiado

git add
    agrega los elementos previo al commit

git commit - m 
    hace commit y agrega mensaje

git diff 'nombre archivo'
    ver los cambios 


git restore --source CODIGO
    volvemos a una version anterior usando el codigo de git log

git branch
    ver las ramas

git checkout -b NOMBRE-RAMA
    crea una nueva ramas

git switch RAMA
    cambiar entre ramas
